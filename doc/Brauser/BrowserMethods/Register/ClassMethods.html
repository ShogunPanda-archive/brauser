def add(type, definition)
  rv = false

  if [:browsers, :platforms, :languages].include?(type) then
    @definitions ||= {}
    @definitions[type] ||= {}

    definition.ensure_array.each do |d|
      if d.is_a?(::Brauser::Definition) then
        @definitions[type][d.tag] = d
        rv = true
      end
    end
  end

  rv
end
